# Пример простого плагина для WC

## Минимальный плагин "Hello World"

Этот пример показывает базовую структуру плагина для Wild Commander.

```asm
;---------------------------------------
; Hello World Plugin
; Простейший пример плагина для WC
;---------------------------------------

        ORG #6000

; Константы WC API
WLD     EQU #6006       ; Точка вызова API
ABT     EQU #6004       ; Флаг ESC
ENT     EQU #6005       ; Флаг ENTER

;---------------------------------------
; ЗАГОЛОВОК ПЛАГИНА (512 байт)
;---------------------------------------

; +0: Резерв (16 байт)
        DS 16

; +16: Идентификатор (16 байт)
        DB "WildCommanderMDL"

; +32: Версия формата
        DB #10

; +33: Резерв
        DB 0

; +34: Количество страниц
        DB 1                    ; Плагин занимает 1 страницу

; +35: Стартовая страница
        DB 0                    ; Запускаться будет со страницы 0

; +36: Блоки данных (6 x 2 байта)
        DB 0,1                  ; Блок 0: код, 1x512 байт
        DB 0,0
        DB 0,0
        DB 0,0
        DB 0,0
        DB 0,0

; +48: Резерв (15 байт)
        DS 15

; +63: Флаги расширений
        DB 0                    ; Стандартный режим

; +64: Блок расширений (32 x 3 байта)
        DS 32*3                 ; Не используем расширения

; +160: Конец блока расширений
        DB 0

; +161: Максимальный размер файла (4 байта)
        DW #FFFF,#FFFF         ; Без ограничений

; +165: Имя плагина (32 байта)
        DB "Hello World Plugin           "

; +197: Тип условий вызова
        DB #03                  ; Из меню (F10)

; +198: Текст для F2/F4 (6 байт)
        DS 6

; +204: Текст для меню вьювера (24 байта)
        DS 24

; +228: Резерв (32 байта)
        DS 32

; +260: Параметры INI (252 байта)
        DB 0                    ; Нет параметров
        DS 251

;---------------------------------------
; КОД ПЛАГИНА
;---------------------------------------

        ORG #8000,#6200

;---------------------------------------
; Точка входа плагина
;---------------------------------------
PLUGIN:
        ; Сохранить входные параметры
        LD (SAVE_IX),IX
        LD (SAVE_HL),HL
        LD (SAVE_DE),DE
        LD (SAVE_BC),BC
        LD (SAVE_A),A

        ; Восстановить текстовый режим
        CALL GEDPL

        ; Показать окно
        LD IX,WINDOW
        CALL PRWOW

        ; Вывести текст
        LD HL,TEXT
        LD DE,#0102            ; Y=1, X=2
        LD BC,12               ; Длина
        CALL PRSRW

        ; Применить цвет
        EXA
        LD A,%00001111         ; Синий на белом
        CALL PRIAT
        EXA

; Главный цикл
MAIN_LOOP:
        EI
        HALT                   ; Ждем прерывание

        ; Проверка ESC
        CALL ESC_KEY
        JR Z,MAIN_LOOP

        ; Выход
        LD IX,WINDOW
        CALL RRESB             ; Стереть окно

        XOR A                  ; A=0: просто выход
        RET

;---------------------------------------
; WC API функции
;---------------------------------------

PRWOW:  ; Вывод окна
        LD A,1
        JP WLD

RRESB:  ; Стирание окна
        LD A,2
        JP WLD

PRSRW:  ; Печать строки
        LD A,3
        JP WLD

PRIAT:  ; Установка цвета
        EXA
        LD A,4
        JP WLD

ESC_KEY: ; Проверка ESC
        LD A,#17
        JP WLD

GEDPL:  ; Восстановление TXT режима
        LD A,#0F
        JP WLD

;---------------------------------------
; Данные
;---------------------------------------

; Структура окна
WINDOW:
        DB 0                   ; Флаги
        DB 0                   ; Маска курсора
        DB 255                 ; X (центр)
        DB 255                 ; Y (центр)
        DB 30                  ; Ширина
        DB 5                   ; Высота
        DB %01111111           ; Цвет (белый на синем)
        DB 0                   ; Резерв
        DW #0000               ; Буфер
        DB 0,0                 ; Разделители

; Текст
TEXT:
        DB "Hello World!"

; Сохраненные регистры
SAVE_A:  DB 0
SAVE_BC: DW 0
SAVE_DE: DW 0
SAVE_HL: DW 0
SAVE_IX: DW 0

        END
```

## Компиляция

Используйте sjasm для компиляции:

```batch
sjasm hello.asm hello.wmf
```

## Пояснения к коду

### Структура

1. **Заголовок (#6000-#61FF)** - 512 байт метаданных
2. **Код (#8000+)** - исполняемый код плагина

### Ключевые моменты

1. **GEDPL обязателен!** Вызывайте его в начале, чтобы восстановить текстовый режим.

2. **Структура окна** должна быть в памяти плагина (#8000-#BFFF).

3. **Главный цикл** должен включать:
   - `EI` - разрешить прерывания
   - `HALT` - ждать прерывание (экономит энергию)
   - Проверку клавиш

4. **Выход** через `RET` с кодом в A.

## Пример с файлом

```asm
;---------------------------------------
; Text Viewer Plugin - просмотр текстовых файлов
;---------------------------------------

        ORG #6000

WLD     EQU #6006

; ... Заголовок плагина ...

; +64: Расширения
        DB "TXT"               ; .txt файлы
        DB "DOC"               ; .doc файлы
        DS 30*3

; +197: Тип вызова
        DB #00                 ; По расширению

        ORG #8000,#6200

;---------------------------------------
; Точка входа
;---------------------------------------
PLUGIN:
        ; Проверить тип вызова
        CP #00                 ; По расширению?
        JR Z,VIEW_FILE
        
        ; Другой тип вызова - не наш файл
        LD A,1                 ; Файл не опознан
        RET

VIEW_FILE:
        ; Сохранить параметры
        LD (FILESIZE),HL
        LD (FILESIZE+2),DE
        LD (FILENAME),BC
        LD (PANEL_IX),IX

        ; Восстановить TXT режим
        CALL GEDPL

        ; Загрузить начало файла
        LD HL,BUFFER
        LD B,16                ; 16 блоков = 8KB
        CALL LOAD512

        ; Показать окно
        LD IX,WINDOW
        CALL PRWOW

        ; Вывести текст
        LD HL,BUFFER
        LD DE,#0101
        LD BC,1000
        CALL PRSRW

; Цикл просмотра
VIEW_LOOP:
        EI
        HALT

        CALL ESC_KEY
        JR Z,VIEW_LOOP

        ; Выход
        LD IX,WINDOW
        CALL RRESB
        XOR A
        RET

;---------------------------------------
; WC API
;---------------------------------------

LOAD512: LD A,#30
         JP WLD

PRWOW:   LD A,1
         JP WLD

RRESB:   LD A,2
         JP WLD

PRSRW:   LD A,3
         JP WLD

ESC_KEY: LD A,#17
         JP WLD

GEDPL:   LD A,#0F
         JP WLD

;---------------------------------------
; Данные
;---------------------------------------

WINDOW:
        DB 0
        DB 0
        DB 0                   ; X=0
        DB 0                   ; Y=0
        DB 80                  ; На весь экран
        DB 25
        DB %01111111
        DB 0
        DW #0000
        DB 0,0

BUFFER:  DS 8192               ; Буфер для файла

FILENAME:   DW 0
FILESIZE:   DS 4
PANEL_IX:   DW 0

        END
```
